CREATE TABLE IF NOT EXISTS tabela_cadastro (
            id INT AUTO_INCREMENT PRIMARY KEY,
            nome VARCHAR(50),
            sobrenome VARCHAR(50),
            idade INT,
            email VARCHAR(50),
            celular VARCHAR(20),
            pais VARCHAR(50),
            estado VARCHAR(50),
            whats TINYINT(1));
